Vary{
  Parameter{   // building azimuth
    Name    = azimuth;
    Min     = SMALL;
    Ini     =     0;
    Max     =   BIG;
    Step    =    10;
  }
  Parameter{    // window and overhang width
    Name    = w_we_win;
    Min     = 0.1;
    Ini     = 3;
    Max     = 5.9;
    Step    = 0.2;
  }
  Parameter{
    Name    = w_ea_win;
    Min     = 0.1;
    Ini     = 3;
    Max     = 5.9;
    Step    = 0.2;
  }
  Parameter{ // solar, visible, and thermal transmittance of shading device
    Name    = tau;
    Min     = 0.2;
    Ini     = 0.5;
    Max     = 0.8;
    Step    = 0.1;
  }
}

  MaxEqualResults = 100;
  WriteStepNumber = false;
  UnitsOfExecution = 0;
}

  Main = GPSHookeJeeves;
  MeshSizeDivider = 2;
  InitialMeshSizeExponent = 0;
  MeshSizeExponentIncrement = 1;
  NumberOfStepReduction = 4;
